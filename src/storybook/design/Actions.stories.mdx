## Actions

export const ActionItem = (props) => {
  const {palette} = useTheme()
  const {name} = props
  return (
    <ColorItem
      title={name}
      colors={{
        "": palette.action[name],
      }}
    />
  )
}

export const ActionsPalette = () => {
  const {palette} = useTheme()
  return (
    <ColorPalette>
      <ActionItem name="active"/>
      <ActionItem name="disabled"/>
      <ActionItem name="hover"/>
      <ActionItem name="focus"/>
      <ColorItem
        title="selected"
        colors={{
          "": alpha(palette.action.selected, palette.action.selectedOpacity),
        }}
      />
    </ColorPalette>
  )
}

### Demo

export const Actionable = () => {
  const [selected, setSelected] = useState(false)
  return (
    <ListItem>
      <ListItemButton
        selected={selected}
        onClick={() => setSelected(selected => !selected)}
      >
        <ListItemIcon>
          <Checkbox checked={selected}/>
        </ListItemIcon>
        <ListItemText>
          Click Me!
        </ListItemText>
      </ListItemButton>
    </ListItem>
  )
}
